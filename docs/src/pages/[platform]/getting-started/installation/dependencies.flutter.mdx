import { Link, Alert, Text, Tabs, TabItem } from '@aws-amplify/ui-react';

You will need to add dependencies for the `amplify_authenticator` to the `pubspec.yaml` file as shown below.

<Alert variation="info">
  <Text color="inherit">
    Web and desktop support for the Amplify Flutter Auth category is now in developer preview! See the [Auth category documentation](https://docs.amplify.aws/lib/auth/getting-started/q/platform/flutter/) for more information about getting started.
  </Text>
</Alert>

<Tabs justifyContent="flex-start">
  <TabItem title="Stable (Mobile)">
    ```yaml
      dependencies:
        amplify_flutter: ^0.6.0
        amplify_auth_cognito: ^0.6.0
        amplify_authenticator: ^0.2.0
    ```
  </TabItem>
  <TabItem title="Developer Preview (Mobile, Web & Desktop)">
    ```yaml
      dependencies:
        amplify_flutter: ^1.0.0-0
        amplify_auth_cognito: ^1.0.0-0
        amplify_authenticator: ^1.0.0-0
    ```
  </TabItem>
</Tabs>
